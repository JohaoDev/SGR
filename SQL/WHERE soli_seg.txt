select S.ROWID, 
S.SOLI_SEC,
S.PERS_SEC,
S.SOLI_ASU,
S.SOLI_DES,
S.SOLI_NUM,
S.SOLI_TIP,
S.SOLI_EST,
S.SOLI_BAS_CON,
S.SOLI_SOL,
S.SOLO_PRI,
S.SOLO_IMP,
S.SOLI_EXT,
S.SOLI_VNC,
S.SOLI_FEC,
S.SOLI_FEC_FIN,
S.ESTR_SEC_ORI,
S.ESTR_SEC_DES,
S.REQU_SEC
from SOLI S,
     SOLI_SEG T
WHERE S.SOLI_SEC = T.SOLI_SEC AND
      S.PERS_SEC = :VG_PERS_SEC
